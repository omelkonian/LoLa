  	\begin{frame}{G$_2$: Adding states}
   	\small
   	\begin{flalign*}
 		&\left . \ \begin{array}{lr}
					\Or{A^+}{\epsilon}{a}{2} \\
					\Or{B^+}{\epsilon}{b}{2} \\
					\Or{C^+}{\epsilon}{c}{2}
           \end{array}
           \right\} \tsc{base cases} \\
  		&\noindent\rule{11cm}{0.4pt}\\
 		&\left . \ \begin{array}{lr}
					\Or{C^-}{A^+, B^+}{x < y < z < w}{2} \\
					\Or{B^-}{A^+, C^+}{x < y < z < w}{2} \\
					\Or{A^-}{B^+, C^+}{x < y < z < w}{2} \\
					\Or{A^+}{C^-, B^-}{x < y < z < w}{2} \\
					\Or{B^+}{C^-, A^-}{x < y < z < w}{2} \\
					\Or{C^+}{B^-, A^-}{x < y < z < w}{2} \\
					\forall \ \s{K} \in \mathcal{S} \setminus \s{W}:\ \Or{\s{K}}{\s{K}_{xy}, \s{W}_{zw}}{x < y,\ z < w}{2}
           \end{array}
           \right\} \tsc{combinations} \\
		&\noindent\rule{11cm}{0.4pt}\\
  		&\left . \ \begin{array}{lr}
					\Or{W}{A^+, A^-}{x < y < z < w}{2} \\
					\Or{W}{C^-, C^+}{x < y < z < w}{2}
           \end{array}
           \right\} \tsc{closures}
	\end{flalign*}
  	\end{frame}
